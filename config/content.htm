# 從 Brython 程式庫中的 browser 導入 html, document, ajax
from browser import html, document, ajax

# 定義處理學員資料的函式
def process_data(req):
    if req.status == 200:  # 確保請求成功
        data = req.text.splitlines()  # 讀取並拆成每一行
        all_stud = data[1:]  # 跳過標題列
        brython_div1 = document["brython_div1"]  # 取得對應的網頁區塊

        # 清空原有內容
        brython_div1.clear()

        for line in all_stud:
            fields = line.split("\t")  # 以 Tab 分隔學號與 GitHub 帳號
            if len(fields) < 2 or not fields[1].strip():  # 若帳號欄位為空，則跳過
                continue
            
            stud_num = fields[0].strip()  # 學號
            github_account = fields[1].strip()  # GitHub 帳號
            scrum_url = f"https://mdecd2025.github.io/hw-{github_account}"  # Scrum 連結
            github_url = f"https://github.com/mdecd2025/hw-{github_account}"  # GitHub 連結

            # 插入學號與 GitHub 連結
            brython_div1 <= html.P(
                [
                    html.A(stud_num, href=scrum_url, target="_blank", style={"margin-right": "10px"}),  # 學號連結
                    " | ",
                    html.A(github_account, href=github_url, target="_blank")  # GitHub 帳號作為超連結
                ]
            )

# 使用 AJAX 非同步讀取 2b.txt
ajax.get("https://mde.tw/list/2b.txt", oncomplete=process_data)

Exam3 (20%): Webots 機電模擬場景的協同設計

各分組利用 CAD (Solvespace 與 NX2312), 隨堂建立指定的系統模型零組件後, 導入 Webots R2025a 後, 建立機電系統模擬場景, 並利用 Python 程式進行互動控制. 過程中各學員必須採同步協同模式, 維護從 Github Classroom 取得的分組協同倉儲以及網站.

協同分組方式:
